4. SQLマイグレーション

Supabase SQL Editorまたは Prisma migrate で実行。

-- ========================================
-- 犬旅日和 DB初期マイグレーション
-- Supabase PostgreSQL
-- ========================================

-- ENUMs
CREATE TYPE hotel_type AS ENUM (
  'hotel', 'ryokan', 'pension', 'cottage',
  'glamping', 'minshuku', 'resort'
);

CREATE TYPE listing_plan AS ENUM (
  'free', 'standard', 'premium'
);

CREATE TYPE booking_provider AS ENUM (
  'rakuten', 'jalan', 'ikyu', 'direct'
);

CREATE TYPE amenity_category AS ENUM (
  'dog_facility', 'general', 'room'
);

CREATE TYPE article_category AS ENUM (
  'area_guide', 'tips', 'ranking', 'news', 'feature'
);

-- ========================================
-- regions (地方マスタ)
-- ========================================
CREATE TABLE regions (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name        varchar(50) NOT NULL,
  slug        varchar(50) NOT NULL UNIQUE,
  sort_order  int NOT NULL DEFAULT 0,
  created_at  timestamptz DEFAULT now()
);

-- ========================================
-- prefectures (都道府県マスタ)
-- ========================================
CREATE TABLE prefectures (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  region_id   uuid NOT NULL REFERENCES regions(id),
  name        varchar(20) NOT NULL,
  slug        varchar(50) NOT NULL UNIQUE,
  lat         decimal(10,7),
  lng         decimal(10,7),
  hotel_count int DEFAULT 0,
  created_at  timestamptz DEFAULT now()
);

-- ========================================
-- hotels (宿メインテーブル)
-- ========================================
CREATE TABLE hotels (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  prefecture_id   uuid NOT NULL REFERENCES prefectures(id),
  name            varchar(200) NOT NULL,
  slug            varchar(200) NOT NULL UNIQUE,
  description     text,
  postal_code     varchar(10),
  address         varchar(300) NOT NULL,
  lat             decimal(10,7),
  lng             decimal(10,7),
  phone           varchar(20),
  website_url     text,
  hotel_type      hotel_type NOT NULL,
  checkin_time    time,
  checkout_time   time,
  room_count      int,
  pet_room_count  int,
  parking         varchar(100),
  access          text,
  listing_plan    listing_plan DEFAULT 'free',
  is_published    boolean DEFAULT true,
  featured_order  int,
  view_count      int DEFAULT 0,
  created_at      timestamptz DEFAULT now(),
  updated_at      timestamptz DEFAULT now()
);

-- フルテキスト検索用インデックス
CREATE INDEX idx_hotels_search ON hotels
  USING GIN (to_tsvector('japanese', name || ' ' || coalesce(description,'') || ' ' || address));

-- slugでの高速検索
CREATE INDEX idx_hotels_slug ON hotels(slug);
CREATE INDEX idx_hotels_prefecture ON hotels(prefecture_id);
CREATE INDEX idx_hotels_published ON hotels(is_published) WHERE is_published = true;

-- ========================================
-- hotel_dog_policies (犬の受入れ条件 1:1)
-- ========================================
CREATE TABLE hotel_dog_policies (
  id                      uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  hotel_id                uuid NOT NULL UNIQUE REFERENCES hotels(id) ON DELETE CASCADE,
  accepted_sizes          text[] NOT NULL,
  max_weight_kg           decimal(5,1),
  max_dogs                int NOT NULL DEFAULT 1,
  breed_restriction       text,
  dog_fee                 int,
  dog_fee_note            text,
  dog_free_in_room        boolean DEFAULT false,
  required_vaccinations   text[],
  dog_areas               text[],
  notes                   text,
  created_at              timestamptz DEFAULT now()
);

-- 大型犬OK / 頭数での検索を高速化
CREATE INDEX idx_dog_policies_sizes ON hotel_dog_policies USING GIN(accepted_sizes);
CREATE INDEX idx_dog_policies_max_dogs ON hotel_dog_policies(max_dogs);

-- ========================================
-- amenities (設備マスタ)
-- ========================================
CREATE TABLE amenities (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name        varchar(100) NOT NULL,
  slug        varchar(100) NOT NULL UNIQUE,
  category    amenity_category NOT NULL,
  icon        varchar(50),
  sort_order  int DEFAULT 0
);

-- ========================================
-- hotel_amenities (宿×設備 中間テーブル)
-- ========================================
CREATE TABLE hotel_amenities (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  hotel_id    uuid NOT NULL REFERENCES hotels(id) ON DELETE CASCADE,
  amenity_id  uuid NOT NULL REFERENCES amenities(id),
  notes       text,
  UNIQUE(hotel_id, amenity_id)
);

-- ========================================
-- hotel_images (宿画像)
-- ========================================
CREATE TABLE hotel_images (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  hotel_id    uuid NOT NULL REFERENCES hotels(id) ON DELETE CASCADE,
  url         text NOT NULL,
  alt_text    varchar(200),
  sort_order  int DEFAULT 0,
  is_main     boolean DEFAULT false
);

-- ========================================
-- booking_links (アフィリエイトリンク)
-- ========================================
CREATE TABLE booking_links (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  hotel_id        uuid NOT NULL REFERENCES hotels(id) ON DELETE CASCADE,
  provider        booking_provider NOT NULL,
  affiliate_url   text NOT NULL,
  is_active       boolean DEFAULT true,
  UNIQUE(hotel_id, provider)
);

-- ========================================
-- hotel_faqs (宿FAQ → JSON-LD FAQPage用)
-- ========================================
CREATE TABLE hotel_faqs (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  hotel_id    uuid NOT NULL REFERENCES hotels(id) ON DELETE CASCADE,
  question    text NOT NULL,
  answer      text NOT NULL,
  sort_order  int DEFAULT 0
);

-- ========================================
-- articles (記事データ)
-- ========================================
CREATE TABLE articles (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  title           varchar(200) NOT NULL,
  slug            varchar(200) NOT NULL UNIQUE,
  description     text,
  content         text NOT NULL,
  category        article_category NOT NULL,
  tags            text[],
  thumbnail_url   text,
  is_published    boolean DEFAULT false,
  published_at    timestamptz,
  created_at      timestamptz DEFAULT now(),
  updated_at      timestamptz DEFAULT now()
);

-- ========================================
-- article_hotels (記事×宿 中間テーブル)
-- ========================================
CREATE TABLE article_hotels (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  article_id  uuid NOT NULL REFERENCES articles(id) ON DELETE CASCADE,
  hotel_id    uuid NOT NULL REFERENCES hotels(id),
  sort_order  int DEFAULT 0,
  UNIQUE(article_id, hotel_id)
);

-- ========================================
-- updated_at自動更新トリガー
-- ========================================
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS trigger AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ language plpgsql;

CREATE TRIGGER hotels_updated_at
  BEFORE UPDATE ON hotels
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER articles_updated_at
  BEFORE UPDATE ON articles
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

-- ========================================
-- RLS (Row Level Security) — Supabase推奨
-- ========================================
ALTER TABLE hotels ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Public read" ON hotels
  FOR SELECT USING (is_published = true);

ALTER TABLE articles ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Public read" ON articles
  FOR SELECT USING (is_published = true);

// lib/queries/hotels.ts
import { supabase } from '@/lib/supabase'

export async function getHotelBySlug(slug: string) {
  const { data } = await supabase
    .from('hotels')
    .select(`
      *,
      prefecture:prefectures(*, region:regions(*)),
      dog_policy:hotel_dog_policies(*),
      amenities:hotel_amenities(*, amenity:amenities(*)),
      images:hotel_images(*),
      booking_links(*),
      faqs:hotel_faqs(*)
    `)
    .eq('slug', slug)
    .eq('is_published', true)
    .single()

  return data
}

export async function getHotelsByPrefecture(
  prefectureSlug: string,
  filters?: {
    sizes?: string[]
    amenities?: string[]
    minDogs?: number
  }
) {
  let query = supabase
    .from('hotels')
    .select(`
      *,
      dog_policy:hotel_dog_policies(*),
      amenities:hotel_amenities(amenity:amenities(name, slug)),
      images:hotel_images(url, alt_text).filter(is_main.eq.true)
    `)
    .eq('is_published', true)

  // 都道府県フィルタ
  if (prefectureSlug) {
    const { data: pref } = await supabase
      .from('prefectures').select('id')
      .eq('slug', prefectureSlug).single()
    if (pref) query = query.eq('prefecture_id', pref.id)
  }

  const { data } = await query.order('featured_order', { ascending: true, nullsFirst: false })
  return data ?? []
}

5-3. JSON-LD生成コンポーネント

// components/seo/HotelJsonLd.tsx
import type { Hotel } from '@/types/hotel'

export function HotelJsonLd({ hotel }: { hotel: Hotel }) {
  const jsonLd = {
    '@context': 'https://schema.org',
    '@type': 'LodgingBusiness',
    name: hotel.name,
    description: hotel.description,
    url: `https://www.inutabi-biyori.jp/hotel/${hotel.slug}`,
    image: hotel.images?.map(img => img.url),
    address: {
      '@type': 'PostalAddress',
      addressRegion: hotel.prefecture?.name,
      streetAddress: hotel.address,
      postalCode: hotel.postal_code,
      addressCountry: 'JP',
    },
    geo: hotel.lat && hotel.lng ? {
      '@type': 'GeoCoordinates',
      latitude: hotel.lat,
      longitude: hotel.lng,
    } : undefined,
    telephone: hotel.phone,
    checkinTime: hotel.checkin_time,
    checkoutTime: hotel.checkout_time,
    numberOfRooms: hotel.room_count,
    petsAllowed: true,
    // 犬用設備を amenityFeature に変換
    amenityFeature: hotel.amenities?.map(ha => ({
      '@type': 'LocationFeatureSpecification',
      name: ha.amenity.name,
      value: true,
    })),
    // 犬特化データを additionalProperty に変換
    additionalProperty: hotel.dog_policy ? [
      {
        '@type': 'PropertyValue',
        name: 'acceptedDogSize',
        value: hotel.dog_policy.accepted_sizes.join(','),
      },
      {
        '@type': 'PropertyValue',
        name: 'maxDogsPerRoom',
        value: String(hotel.dog_policy.max_dogs),
      },
      hotel.dog_policy.max_weight_kg && {
        '@type': 'PropertyValue',
        name: 'maxDogWeight',
        value: String(hotel.dog_policy.max_weight_kg),
        unitCode: 'KGM',
      },
      hotel.dog_policy.dog_fee != null && {
        '@type': 'PropertyValue',
        name: 'dogFeePerNight',
        value: String(hotel.dog_policy.dog_fee),
        unitCode: 'JPY',
      },
      {
        '@type': 'PropertyValue',
        name: 'dogFreeInRoom',
        value: String(hotel.dog_policy.dog_free_in_room),
      },
    ].filter(Boolean) : [],
  }

  return (
    <script
      type="application/ld+json"
      dangerouslySetInnerHTML={{ __html: JSON.stringify(jsonLd) }}
    />
  )
}

5-4. 宿詳細ページ (App Router)

// app/hotel/[slug]/page.tsx
import { notFound } from 'next/navigation'
import { getHotelBySlug } from '@/lib/queries/hotels'
import { HotelJsonLd } from '@/components/seo/HotelJsonLd'
import { FaqJsonLd } from '@/components/seo/FaqJsonLd'
import { BreadcrumbJsonLd } from '@/components/seo/BreadcrumbJsonLd'

// ISR: 1時間ごとに再生成
export const revalidate = 3600

// 全宿ページを事前生成 (SSG)
export async function generateStaticParams() {
  const { data } = await supabase
    .from('hotels').select('slug').eq('is_published', true)
  return data?.map(h => ({ slug: h.slug })) ?? []
}

// メタデータ生成
export async function generateMetadata({ params }: { params: { slug: string }}) {
  const hotel = await getHotelBySlug(params.slug)
  if (!hotel) return {}

  const sizes = hotel.dog_policy?.accepted_sizes?.join('・') ?? ''
  return {
    title: `${hotel.name} | 犬旅日和`,
    description: `${hotel.name}の犬連れ宿泊情報。${sizes}OK、最大${hotel.dog_policy?.max_dogs}頭。${hotel.prefecture?.name}。`,
    openGraph: {
      images: hotel.images?.[0]?.url,
    },
  }
}

// ページコンポーネント
export default async function HotelPage({ params }: { params: { slug: string }}) {
  const hotel = await getHotelBySlug(params.slug)
  if (!hotel) notFound()

  const breadcrumbs = [
    { name: 'トップ', url: '/' },
    { name: hotel.prefecture?.region?.name, url: `/area/${hotel.prefecture?.region?.slug}` },
    { name: hotel.prefecture?.name, url: `/area/${hotel.prefecture?.slug}` },
    { name: hotel.name },
  ]

  return (
    <>
      <HotelJsonLd hotel={hotel} />
      <FaqJsonLd faqs={hotel.faqs} />
      <BreadcrumbJsonLd items={breadcrumbs} />

      <main>
        <Breadcrumb items={breadcrumbs} />
        <HotelDetail hotel={hotel} />
      </main>
    </>
  )

5-5. 検索API Route

// app/api/search/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { supabase } from '@/lib/supabase'

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url)
  const q = searchParams.get('q') ?? ''
  const prefecture = searchParams.get('prefecture')
  const sizes = searchParams.getAll('size')    // 大型犬 etc.
  const amenities = searchParams.getAll('amenity')
  const minDogs = searchParams.get('min_dogs')
  const page = Number(searchParams.get('page') ?? '1')
  const perPage = 20

  let query = supabase
    .from('hotels')
    .select(`
      *, dog_policy:hotel_dog_policies(*),
      amenities:hotel_amenities(amenity:amenities(name,slug)),
      images:hotel_images(url).filter(is_main.eq.true),
      prefecture:prefectures(name,slug)
    `, { count: 'exact' })
    .eq('is_published', true)

  // フリーワード検索
  if (q) {
    query = query.textSearch('name', q, { type: 'websearch' })
  }

  // ページネーション
  const from = (page - 1) * perPage
  query = query.range(from, from + perPage - 1)

  const { data, count, error } = await query
  if (error) return NextResponse.json({ error: error.message }, { status: 500 })

  return NextResponse.json({
    hotels: data,
    total: count,
    page,
    totalPages: Math.ceil((count ?? 0) / perPage),
  })
}
